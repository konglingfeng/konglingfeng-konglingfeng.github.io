int num=0;
int f(int n)
{
    if(n==1)
      return 1;

 }

####5.N皇后递归实现

#include "stdafx.h"
#include "math.h"
#define N  8
int count;
int x[N]={0};
void queen();
int place(int k)
{
    int j;
    for(j=0;j<k;j++)
    {
    if(abs(k-j)==abs(x[j]-x[k])||x[j]==x[k])
        return 0;
    }
   return 1;
}
void queen(int t)
{
    int i;
    if(t==N)
    count++;
    for(i=0;i<N;i++)
    {
    x[t]=i;
    if(place(t))
    {
      queen(t+1);
    }

}
int main()
{
    int num;
    scanf("%d",&num);
    queen(num);
    printf("count=%d\n",count);
    return 0;
}
